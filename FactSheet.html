function histo_smooth(xdata, bins)
    p = length(xdata)
    @assert p > 0 "Input data must not be empty."
    uniform = 1 / p * ones(p)
    k = tower_sample(uniform)
    l = searchsortedfirst(bins, xdata[k]) #warning: this implementation relies on how "bins" is defined
    dist = Uniform(bins[l-1], bins[l])
    x = rand(dist)
    return x
end

function gauss_kernel(xdata, sigma)
    p = length(xdata)
    Pidist = zeros(p)
    for i in 1:p
        Pidist[i] = 1 / p 
    end
    mu = tower_sample(Pidist)
    dist = Normal(xdata[mu], sigma)
    x = rand(dist)
    return x
end

function empirical_cdf(xdata, x)
    p = length(xdata)
    count = 0
    for i in 1:p
        if xdata[i] <= x
            count += 1
        end
    end
    return count / p
end

function histogram_cdf(histogram_data, x)
    bins = histogram_data.edges[1]
    counts = histogram_data.weights
    total = sum(counts)
    cum_count = 0
    for i in 1:length(bins)-1
        if x >= bins[i+1]
            cum_count += counts[i]
        else
            break
        end
    end
    return cum_count / total
end

function forward_DDPM(patterns, beta, N_steps)
    p = length(patterns)
    dist = DiscreteUniform(1, p)
    mu = rand(dist)
    final_target = randn()
    traj = levy_free_path(patterns[mu], final_target, N_steps, beta)
    return traj
end
